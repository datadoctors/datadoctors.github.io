Loyalty Receipt Sample

The following XML sample shows how the loyalty add-in prints loyalty information on receipts.

<SUB name="PrintMembershipStatus">

    <IF>

        <CONDITION>

            ShowMembershipStatus &amp; Len(Session.Variable1)

        </CONDITION>

        <THEN>

            <TABLE>

                <FONT>      "Medium"     </FONT>

                <BORDER>    tbNone      </BORDER>

                <COLUMNHEADER>

                    <ALIGNMENT>    "^~"           </ALIGNMENT>

                    <WIDTH>        PageWidth  </WIDTH>

                    <TEXT>  

                        "Thanks for being a member of the loyalty program"  

                    </TEXT>

                </COLUMNHEADER>

            </TABLE>

  

            <TABLE>

                <FONT>      "VerySmall"     </FONT>

                <BORDER>    tbNone      </BORDER>

                <COLUMNHEADER>

                    <ALIGNMENT>    "^~"           </ALIGNMENT>

                    <WIDTH>        PageWidth </WIDTH>

                    <TEXT>         

                        "* * * * Your Membership Level is " Session.Variable1 " * * * *"

                    </TEXT>

                </COLUMNHEADER>

            

                <ROW> Session.Variable2 </ROW>

            </TABLE>

        </THEN>

        <ELSE>

            <TABLE>

                <FONT>      "Medium"     </FONT>

                <BORDER>    tbNone      </BORDER>

                <COLUMNHEADER>

                    <ALIGNMENT>    "^~"           </ALIGNMENT>

                    <WIDTH>        PageWidth  </WIDTH>

                    <TEXT>  

                        "Enroll now for the loyalty program !!!"  

                    </TEXT>

                </COLUMNHEADER>

            </TABLE>

       </ELSE>

 

      </IF>

   </SUB>
