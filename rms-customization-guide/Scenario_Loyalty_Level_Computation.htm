Loyalty Level Computation Sample

The following COM component sample code shows how the loyalty add-in uses the Process function to compute the loyalty level of the current customer.

Public Function Process(ByVal session As Object) As Boolean

    Try

        Dim custID As Integer

        Dim transactionTotal As Double

        Dim transactionType As Integer

 

        ' Get the current subtotal for all items being purchased

        ' in the transaction.

        transactionTotal = session.Transaction.SubtotalPurchased()

 

        ' Get the customer ID.

        custID = session.Transaction.Customer.ID

 

        ' Get the transaction type.

        transactionType = session.Transaction.TransactionType

 

        ' Compute the loyalty level only for a sales transaction.

        If (transactionType = 1) Then

            ' Update loyaltySales in the database for the customer.

            CalculateLoyalty(custID, transactionTotal)

        End If

 

        ' Set up variables for the Loyalty receipt.

        Dim status As Boolean

 

        ' Initialize the session variables for the receipt.

        session.Variable(1) = ""   ' Loyalty level name.

        session.Variable(2) = ""   ' Purchases required for next level.

 

        ' Determine whether the customer is loaded.

        If session.Transaction.Customer.Loaded Then

            ' Create a connection to the LoyaltyLevel database.

            objRetrieveLoyalty.SetConnection()

            ' Retrieve the customer ID.

            objRetrieveLoyalty.customerID = _

                session.Transaction.Customer.ID

            ' Determine whether the customer is enrolled.

            status = objRetrieveLoyalty.GetLoyaltyStatus()

            ' If the customer is enrolled, retrieve the level name.

            If (status) Then

                ' Retrieve the level name so that it can be accessed

                ' from the reports.

                session.Variable(1) = _

                    objRetrieveLoyalty.loyaltyLevelName

                ' Check if Purchases not equal to zero.

                If Not (objRetrieveLoyalty.purchasesReqd = 0) Then

                    ' Set the variable which holds the purchases

                    ' required for next level.

                    session.variable(2) = "Purchases Reqd for " _

                        & objRetrieveLoyalty.nextLevelName _

                        & " Level is $ " _

                        & objRetrieveLoyalty.purchasesReqd

                End If

            End If

            ' Close the connection.

            objRetrieveLoyalty.CloseConnection()

        End If

        Process = True

 

        Catch exSQL As SqlException

            ' Log the SQL exception if it occurs.

            errorObj.LogEvent(GetValueFromResFile("TITLE_LL_COMP"), _

                exSQL.Message)

 

        Finally

            errorObj = Nothing

 

    End Try

 

End Function
