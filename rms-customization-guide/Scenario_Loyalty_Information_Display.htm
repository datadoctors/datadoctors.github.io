Loyalty Information Display Sample

The following VBScript sample code shows how the loyalty add-in displays the loyalty information for the current customer in the status display.

Function displayVariables

    On Error Resume Next

 

    Dim QSRules

    Dim Entry

    Dim Custom1

    Dim Custom2

    Dim Custom3

    Dim Custom4

    Dim Custom5

    Dim CustomerEnrolled

    Dim status

    Dim messageCode

  

    ' Get the QSRules.

    Set QSRules = qsBridge.RequestQSRules()

 

    styleTitle.style.visibility = "hidden"

    trDots.style.visibility = "hidden"

    txtStoreName.innerText = ""

    Custom1   = ""

    Custom2   = ""

    Custom3   = ""

    Custom4   = ""

    Custom5   = ""

 

    ' Set the status display message code.

    messageCode = MSGCODE

 

    ' Hide the membership status pane.

    Set Entry = Nothing

 

    ' Get the error code.

    If (errorOccurred = True) Then

        ' If an error occurred, stop the process.

        Exit Function

    End If

 

    If Not QSRules Is Nothing Then     

        ' Check whether the customer is loaded.

        If QSRules.Transaction.Customer.Loaded Then

            ' If the customer is loaded, set the connection

            ' and check for enrollment.

             status = Loyalty.SetConnection()

            ' Set the customer ID.

            Loyalty.customerID = QSRules.Transaction.Customer.ID

            ' Determine whether the customer is enrolled.

            CustomerEnrolled = Loyalty.GetLoyaltyStatus()

 

            ' If the customer is enrolled, display the membership

            ' status, otherwise display the sign-up prompt.

            If (CustomerEnrolled = False) Then

                ' Get the status display message.

                EnrollMessage.innerText = _

                    Loyalty.GetMessage(messageCode) _

                    styleTitle.style.visibility = "hidden"

                trDots.style.visibility = "hidden"

                txtStoreName.innerText = ""

            Else

                ' If the customer is enrolled, no message is displayed

                ' in the status pane.

                ' Hide the enrollment UI.

                styleTitle.style.visibility = "visible"

                trDots.style.visibility = "visible"

 

                EnrollMessage.innerText = ""

                txtStoreName.innerText = "Loyalty Program Status"

                txtCustomLabel1.innerText = "Date Joined:"

                txtCustomLabel2.innerText = "Current Level:"

                txtCustomLabel3.innerText = "Next Level:"

                txtCustomLabel4.innerText = "Purchases Req'd:"

                txtCustomLabel5.innerText = "Discount:"

                Custom1 = Loyalty.LoyaltyJoined

                Custom2 = Loyalty.LoyaltyLevelName

                Custom3 = Loyalty.NextLevelName

                Custom4 = "$" & Loyalty.PurchasesReqd

                Custom5 = Loyalty.discountPercentage & " %"

            End If

 

            Loyalty.CloseConnection()

 

        End If

 

    End If

 

    txtCustom1.innerText = Custom1

    txtCustom2.innerText = Custom2

    txtCustom3.innerText = Custom3

    txtCustom4.innerText = Custom4

    txtCustom5.innerText = Custom5

 

    errorOccurred = Loyalty.errOccurred

 

End Function
