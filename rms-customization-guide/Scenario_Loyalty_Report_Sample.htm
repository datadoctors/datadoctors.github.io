Loyalty Report Sample

The following sample report summary shows how the loyalty effectiveness report queries both the RMS database and the loyalty database to retrieve information.

Begin ReportSummary

    ReportType = reporttypeCustomer

    ReportTitle = "Loyalty Effectiveness Report"

    PageOrientation = pageorientationLandscape

    OutLineMode = True

    WordWrap = True

    Groups = 0

    GroupDescription = "Total"

    DisplayLogo = True

    LogoFileName = "MyLogo.bmp"

    ProcedureCall = ""

 

    PreQuery1 = "IF EXISTS (

        SELECT * from loyalty.dbo.sysobjects

        WHERE id = object_id(N'Loyalty.[dbo].[TempLoyalty]'))

        DROP table Loyalty.dbo.TempLoyalty"

 

    PreQuery2 =

        <BEGIN>

        SELECT

        C.LastName + ', ' + C.FirstName CustomerName,

        C.AccountOpened,

        LS.LoyaltyJoinedDate,

        LL.LoyaltyLevelName,

        CASE DateDiff(Dd,C.AccountOpened, getDate())

            WHEN 0 THEN 1

            ELSE CAST(DateDiff(Dd,C.AccountOpened, getDate()) as real)

            * 12/365 end MonthsPrior,

        CASE DateDiff(Dd, LS.LoyaltyJoinedDate, getDate())

            WHEN 0 THEN 1

            ELSE Cast(DateDiff(Dd, LS.LoyaltyJoinedDate, getDate())

            as real) * 12/365 end MonthsAfter,

        Sum(T.Total - T.SalesTax) - LS.LoyaltySales PurchBeforeJoin,

        (LS.LoyaltySales) PurchAfterJoin

   

        INTO Loyalty.dbo.TempLoyalty

    

        FROM Loyalty.dbo.LoyaltySignUp LS

        LEFT OUTER JOIN Loyalty.dbo.LoyaltyLevel LL

        ON LS.LoyaltyLevelID = LL.LoyaltyLevelID

        INNER JOIN Customer C

        ON C.id = LS.CustomerID

        INNER JOIN [Transaction] T

        ON T.CustomerID = LS.CustomerID

    

        GROUP BY

        C.LastName + ', ' + C.FirstName,

        C.AccountOpened,

        LS.LoyaltyJoinedDate,

        LL.LoyaltyLevelName,

        LS.LoyaltySales

        <END>

 

    TablesQueried = "FROM Loyalty.dbo.TempLoyalty"

    SelCriteria = ""

    GroupBy = ""

    SortOrder = ""

End ReportSummary
